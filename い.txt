myproject/                         # 🌱 プロジェクトのルートディレクトリ
│
├── data/                         # 📂 データ保存用（分析対象のファイルを格納）
│   ├── raw/                      # └📦 生データ（外部から入手した加工前のデータ）
│   ├── processed/                # └📦 加工済みデータ（前処理後に保存したCSVなど）
│   └── external/                 # └📦 外部データ（ベンチマーク・外部リファレンスなど）
│   # ※ Gitで管理しないよう .gitignore に含めておくと安全！
│
├── notebooks/                    # 📒 Jupyter Notebook 置き場（EDAや試行錯誤用）
│   └── analysis_2025_04_12.ipynb # └💡 データ分析・可視化・アイデアメモなどを記録
│
├── myproject/                    # 📦 Pythonパッケージの本体（import対象のモジュール群）
│   ├── __init__.py               # └📌 パッケージであることを示す（空でOK）
│   ├── features/                 # 📂 特徴量エンジニアリング関連のコード
│   │   └── build_features.py     # └🛠 特徴量を生成する関数群
│   ├── models/                   # 📂 モデルの定義・学習・推論用スクリプト
│   │   ├── train_model.py        # └📈 モデルの学習ロジック
│   │   └── predict_model.py      # └🔮 モデルによる予測
│   ├── data/                     # 📂 データ読み込みや保存のユーティリティ
│   │   └── load_data.py          # └📥 CSVやExcel読み込み処理など
│   ├── utils.py                  # 🔧 共通で使えるユーティリティ関数（ログ・時間計測など）
│   └── main.py                   # 🚀 メインスクリプト（学習・評価の起点など）
│
├── scripts/                      # 🧾 実行用スクリプト（CLIで回すバッチ処理など）
│   └── train.py                  # └🚀 `python scripts/train.py` で実行されるモデル訓練スクリプト
│
├── tests/                        # 🧪 テストコード用ディレクトリ（pytestで実行）
│   ├── __init__.py
│   └── test_utils.py             # └✅ ユーティリティ関数の動作確認など
│
├── pyproject.toml                # 📋 Poetryによる依存関係・メタデータ設定ファイル
├── README.md                     # 📖 プロジェクト概要や使い方を説明するMarkdown文書
└── .gitignore                    # 🚫 Gitで追跡しないファイル・フォルダを定義（例: data/, .venv/ など）
